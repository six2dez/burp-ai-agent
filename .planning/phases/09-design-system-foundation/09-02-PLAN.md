    formGrid(): JPanel
      - JPanel(GridBagLayout()); background = Colors.surface; border = EmptyBorder(0, 0, Spacing.formGridPad, 0)
      - putClientProperty("row", 0); return panel

    addRowFull(grid: JPanel, labelText: String, field: JComponent, helpText: String? = null)
      - row = nextRow(grid)
      - label GBC: gridx=0, gridy=row, anchor=WEST, insets=Spacing.rowInsets
      - label: JLabel(labelText), font=Typography.body, foreground=Colors.onSurface
      - field GBC: gridx=1, gridy=row, gridwidth=3, weightx=1.0, insets=Spacing.fieldInsets
      - isSmallComponent check: JSpinner || JComboBox<*> || JCheckBox || ToggleSwitch || (JTextField && columns<=20)
        true -> anchor=WEST, fill=NONE; false -> fill=HORIZONTAL
      - if helpText != null: helpRow = nextRow(grid); help GBC gridx=1, gridy=helpRow, gridwidth=3,
        weightx=1.0, insets=Insets(0,0,Spacing.xs,0); add helpLabel(helpText) with that GBC

    addRowPair(grid: JPanel, leftLabel: String, leftField: JComponent, rightLabel: String, rightField: JComponent)
      - Mirrors SettingsPanel logic but snaps insets to Spacing constants:
        label insets: Spacing.rowInsets; left field insets: Spacing.fieldPairInsets;
        right label insets: Spacing.rowInsets; right field insets: Spacing.fieldInsets.
      - Same isSmallComponent detection for each field.

    addSpacerRow(grid: JPanel, height: Int = Spacing.xs)
      - nextRow(grid); GBC gridx=0, gridwidth=4, weightx=1.0, fill=HORIZONTAL, insets=Insets(0,0,0,0)
      - add Box.createRigidArea(Dimension(0, height))

    sectionPanel(title: String, subtitle: String, content: JComponent): JPanel
      - header = JPanel(BoxLayout Y_AXIS), background=Colors.surface
      - titleLabel: JLabel(title), font=Typography.sectionTitle, foreground=Colors.onSurface
      - subtitleLabel: JLabel(subtitle), font=Typography.body, foreground=Colors.onSurfaceVariant
      - header.add(titleLabel); header.add(Box.createRigidArea(Dimension(0, Spacing.xs))); header.add(subtitleLabel)
      - return JPanel(BorderLayout()).apply {
          background=Colors.surface; border=EmptyBorder(Spacing.sectionPad, Spacing.sectionPad, Spacing.sectionPad, Spacing.sectionPad)
          add(header, BorderLayout.NORTH); add(content, BorderLayout.CENTER) }

    helpLabel(text: String): JLabel
      - JLabel(text), font=Typography.caption, foreground=Colors.onSurfaceVariant

    primaryButton(label: String): JButton
      - JButton(label).apply { font=Typography.body; background=Colors.primary; foreground=Colors.onPrimary; isBorderPainted=false; isFocusPainted=true; isOpaque=true }

    secondaryButton(label: String): JButton
      - JButton(label).apply { font=Typography.body; background=Colors.surface; foreground=Colors.primary; border=LineBorder(Colors.border, 1, true) }

    buildTabPanel(sections: List<JComponent>): JScrollPane
      - content = JPanel(BoxLayout Y_AXIS); content.background=Colors.surface
      - content.border=EmptyBorder(Spacing.lg, Spacing.lg, Spacing.lg, Spacing.lg)
      - sections.forEachIndexed: add section; if not last add Box.createRigidArea(Dimension(0, Spacing.sm))
      - JScrollPane(content).apply { border=EmptyBorder(0,0,0,0); viewport.background=Colors.surface }

    toolBadge(label: String, style: BadgeStyle): JLabel
      - Create an anonymous JLabel subclass (or inner class named BadgeLabel) that overrides paintComponent:
          val bgColor = when(style) { NATIVE -> Colors.badgeNative; FULL -> Colors.badgeFull }
          val fgColor = when(style) { NATIVE -> Colors.statusSuccess; FULL -> Colors.onSurfaceVariant }
        In paintComponent(g): cast to Graphics2D, set antialias hint, setColor(bgColor), fillRoundRect(0,0,width,height,6,6), super.paintComponent(g)
        isOpaque = false; font=Typography.caption; foreground=fgColor
        border=EmptyBorder(2,6,2,6); text=label

    applyFieldStyle(field: JTextField)
      - field.font=Typography.mono; field.border=LineBorder(Colors.border,1,true)
      - field.background=Colors.inputBackground; field.foreground=Colors.inputForeground

    applyAreaStyle(area: JTextArea)
      - area.font=Typography.mono; area.border=LineBorder(Colors.border,1,true)
      - area.background=Colors.inputBackground; area.foreground=Colors.inputForeground
      - area.lineWrap=true; area.wrapStyleWord=true
